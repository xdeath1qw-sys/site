✅ ШАГ 1: СОЗДАТЬ BIN НА JSONBIN.IO (30 СЕКУНД)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Открыть: https://jsonbin.io

2. Нажать большую зелёную кнопку:
   "CREATE YOUR FIRST BIN"

3. В поле "Your JSON here..." вставить:

   {"pl_users":[],"pl_players":[],"pl_teams":[],"pl_news":[],"pl_tournaments":[],"pl_matches":[],"pl_vetos":[]}

4. Нажать "CREATE"

5. Скопировать ID из URL:
   
   URL будет: https://jsonbin.io/67abc123def456
                             ^^^^^^^^^^^^^^^^
                             ЭТО ТВОЙ BIN ID
   
   Пример ID: 67abc123def456

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ ШАГ 2: ВСТАВИТЬ В КОД (10 СЕКУНД)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Открыть файл: js/storage.js

2. Найти строку 18:
   const JSONBIN_BIN_ID = 'YOUR_BIN_ID_HERE';

3. Заменить на твой ID:
   const JSONBIN_BIN_ID = '67abc123def456';

4. Сохранить (Ctrl+S)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ ШАГ 3: ЗАГРУЗИТЬ НА ХОСТИНГ (10 СЕКУНД)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Загрузить js/storage.js на Netlify

2. Дождаться деплоя

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ ШАГ 4: ПРОВЕРИТЬ (5 СЕКУНД)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Открыть сайт

2. F12 (консоль)

3. Должно быть:
   ────────────────────────────────────────
   [DB] 🌐 Инициализация публичной синхронизации...
   [DB] 🔄 Загрузка из JSONBin...
   [DB] ⬇️ Данные загружены из публичной БД
   [DB] 🌐 pl_users: 0 записей
   [DB] ✅ Синхронизация завершена
   [DB] 🔔 Автообновление каждые 30 секунд
   ────────────────────────────────────────

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎯 ТЕСТ - ПРОВЕРКА СИНХРОНИЗАЦИИ:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. ЗАРЕГИСТРИРУЙ ПОЛЬЗОВАТЕЛЯ:
   ▸ Открой страницу регистрации
   ▸ Зарегистрируй тестового пользователя
   
   В консоли должно появиться:
   [DB] ☁️ Синхронизация pl_users...
   [DB] ✅ pl_users синхронизирован для всех
   ✓ Данные синхронизированы

2. ОТКРОЙ САЙТ В ДРУГОЙ ВКЛАДКЕ:
   ▸ Откройконкогнито/инкогнито режим
   ▸ Открой свой сайт
   ▸ Подожди 30 секунд
   ▸ Страница перезагрузится
   ▸ Новый пользователь ПОЯВИТСЯ! ✅

3. СОЗДАЙ НОВОСТЬ В АДМИН ПАНЕЛИ:
   ▸ Войди как admin
   ▸ Создай новость
   
   В консоли:
   [DB] ☁️ Синхронизация pl_news...
   [DB] ✅ pl_news синхронизирован для всех
   
   ▸ Все пользователи увидят новость через 30 сек! ✅

В: Нужна регистрация?
О: НЕТ! Просто создаёшь BIN анонимно.

В: Данные видны всем?
О: ДА, это публичная статистика (как на HLTV).

В: Могут ли пользователи менять данные?
О: ДА, все могут регистрироваться - регистрации публичные!

В: Как часто обновляются данные?
О: Запись МГНОВЕННО, чтение каждые 30 сек.

В: Лимиты?
О: 100KB и 10,000 запросов/месяц - хватит!
